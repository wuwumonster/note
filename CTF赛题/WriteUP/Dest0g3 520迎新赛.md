    if (config.find(i) == -1) and (config.lower().find(i) != 0):
        result += "config|string()|truncate("+str(config.lower().find(i)+1)+",True,app|string())|last|lower%2b"
    if config.find(i) != -1:
        result += "config|string()|truncate("+str(config.find(i)+1)+",True,app|string())|last%2b"

print(result)
